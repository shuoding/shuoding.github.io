Python list ring


def make_ring(n: int) -> list:
    nodes = [[] for i in range(n)]
    for i in range(n):
        nodes[i].append(nodes[(i + 1) % n])
    return nodes[0]

C++ template metaprogramming


#include <iostream>

template <unsigned N> struct fib {
    static constexpr unsigned val = fib<N - 1>::val + fib<N - 2>::val;
};

template <> struct fib<2> {
    static constexpr unsigned val = 1;
};

template <> struct fib<1> {
    static constexpr unsigned val = 1;
};

int main() {
    std::cout << fib<6>::val << std::endl;
}
